# 1. 소스 코드가 들어갈 경로 설정 및 청소
set(SOURCE_PATH "${CURRENT_BUILDTREES_DIR}/src/${PORT}")
file(REMOVE_RECURSE "${SOURCE_PATH}")
file(MAKE_DIRECTORY "${SOURCE_PATH}")

# 2. 시스템 Git 찾기
find_program(GIT NAMES git git.cmd)

# 3. [핵심] Raw Git Clone 실행 (압축 해제 X, Fetch 제한 우회)
message(STATUS "Executing Raw Git Clone...")
execute_process(
    COMMAND "${GIT}" clone https://github.com/zeux/pugixml.git "${SOURCE_PATH}"
    RESULT_VARIABLE GIT_CLONE_RESULT
)

if(NOT GIT_CLONE_RESULT EQUAL 0)
    message(FATAL_ERROR "Git Clone Failed!")
endif()

# 4. 특정 버전(v1.15)으로 체크아웃
message(STATUS "Checking out v1.15...")
execute_process(
    COMMAND "${GIT}" checkout v1.15
    WORKING_DIRECTORY "${SOURCE_PATH}"
)

vcpkg_check_features(
    OUT_FEATURE_OPTIONS FEATURE_OPTIONS
    FEATURES
        compact PUGIXML_COMPACT
)

vcpkg_cmake_configure(
    SOURCE_PATH "${SOURCE_PATH}"
    OPTIONS
        -DPUGIXML_BUILD_TESTS=OFF
        ${FEATURE_OPTIONS}
)

vcpkg_cmake_install()
vcpkg_copy_pdbs()
vcpkg_cmake_config_fixup(CONFIG_PATH lib/cmake/${PORT})
vcpkg_fixup_pkgconfig()

file(REMOVE_RECURSE "${CURRENT_PACKAGES_DIR}/debug/include")

vcpkg_install_copyright(FILE_LIST "${SOURCE_PATH}/LICENSE.md")
